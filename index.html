import pandas as pd # Read df = pd.read_excel("big_data.xlsx", sheet_name="RawData", engine="openpyxl") # Most common transformations df = df[df['Status'] == 'Active'] df['Margin %'] = (df['Selling Price'] - df['Cost Price']) / df['Selling Price'] * 100 df['Margin %'] = df['Margin %'].round(1) # Write (fastest method) with pd.ExcelWriter("Processed_Report.xlsx", engine='openpyxl') as writer: df.to_excel(writer, sheet_name="Cleaned", index=False) # You can add more sheets easily df.groupby('Category').sum().to_excel(writer, sheet_name="Summary")
